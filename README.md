## Install

```bash
npm install --save react-browser-frame
```

## Usage

```tsx
import * as React from 'react'

import MyComponent from 'react-browser-frame'

class Example extends React.Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
